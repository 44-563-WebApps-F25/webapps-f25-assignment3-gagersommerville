This Node.js id used to read and parse a CSV file. 

 
const fileSystem = require("fs");
const { parse } = require("csv-parse");

const readCSV = filePath => {
	fileSystem.createReadStream(filePath)
		.pipe(parse({
			delimiter: ",",
			from_line: 2
		}))
		.on("data", function(row) {
			console.log(row);
		})
}
